"6px" : "8px"; /* ================= VIGNETTE STRENGTH ================= */ document.body.style.setProperty( "--horror", state.intensity.toFixed(3) ); requestAnimationFrame(loop); } loop(); /* ================= RESET ================= */ logo.addEventListener("click", () => { deactivate(); state.intensity = 0; }); });
